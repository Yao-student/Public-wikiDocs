async function basefetch(URL) {
const response = await fetch(URL);
return await response.json();
}

function getFolderJsonPath(filePath) {
const directory = filePath.substring(0, filePath.lastIndexOf('/'));
const folderJsonPath = `/${directory}/Folder.json`;
return folderJsonPath;
}

document.onreadystatechange = function () {
if (document.readyState == "uninitalized") {
document.getElementById("loader").style.width = "10%"
}
if (document.readyState == "loading") {
document.getElementById("loader").style.width = "40%"
}
if (document.readyState == "interactive") {
document.getElementById("loader").style.width = "60%"
}
if (document.readyState == "complete") {
document.getElementById("loader").style.width = "100%"
}
}

const accelerate = "https://gh.llkk.cc/https://";

const params = new URLSearchParams(window.location.search);

const repo = 'Yao-student/Public-wikiDocs';
const filePath = `${params.get('p') || '404'}.md`;

const localesUrl = `${accelerate}raw.githubusercontent.com/${repo}/main/locales.json`;
const pre_locales = params.get("l");
let locales = "zh-CN";

basefetch(localesUrl)
.then(data => {
if (pre_locales in data["locales"]) {
locales = pre_locales;
} else {
window.location.href = `?p=${params.get('p')}&l=zh-CN`;
}
})
.then(data => {
const apiUrl = `${accelerate}raw.githubusercontent.com/${repo}/main/${locales}/${filePath}`;
const folderJsonPath = getFolderJsonPath(filePath);
const folderJsonUrl = `${accelerate}raw.githubusercontent.com/${repo}/main/${locales}${folderJsonPath}`;

// List 创建
basefetch(folderJsonUrl)
.then(data => {
document.title = data[`${filePath}`] || 'Notanote 文档';
var container = document.getElementById('List');
container.innerHTML = '';
if (data[Page]) {
data[Page].forEach(function (item) {
if (item.Class === 'ClassText') {
var p = document.createElement('p');
p.className = item.Class;
p.textContent = item.Content;
container.appendChild(p);
} else if (item.Class === 'Object') {
var a = document.createElement('a');
a.className = item.Class;
a.href = item.Href || '#';
a.textContent = item.Content;

if (item.SubObject) {
var tag = document.createElement('tag');
tag.className = item.Class + '-SubTitle';
tag.textContent = item.SubObject;
a.appendChild(tag);
}

container.appendChild(a);
}
});
} else {
console.error("未找到与Page常量对应的列表");
}
})

fetch(apiUrl)
.then(response => {
if (!response.ok) {
if (response.status === 404) {
window.location.href = '/404';
throw new Error('404状态，已跳转');
} else {
throw new Error('网络错误');
}
}
return response.text();
})
.then(content => {
const htmlContent = marked.parse(content);
document.getElementById('markdown-container').innerHTML = htmlContent;
})
.catch(error => {
//window.location.href = '/404';
});
});
